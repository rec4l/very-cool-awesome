The canvas renders at 60fps via requestAnimationFrame, but since the server only sends state 30 times per second, positions snap to a new value 30 times per second — which is exactly the choppiness you're seeing.

Two options:

Raise the tick rate to 60 — simple one-line change, doubles server broadcast frequency. Works, but doesn't scale and doesn't teach you anything interesting.
Client-side interpolation — client keeps the last two received states and smoothly interpolates between them on every render frame. Movement becomes perfectly fluid even at 30 ticks/s. This is how real multiplayer games handle it (the client always renders slightly in the past, ~33ms, but you'd never notice).
Option 2 is the right call — it's a genuinely interesting technique, fits the "difficult thing with AI" brief, and keeps server load low. The only tradeoff is a small fixed lag of one tick interval (~33ms), which is imperceptible.

function createPhysics(map) {
  const engine = Engine.create({ gravity: { x: 0, y: 0 } });
  const world = engine.world;

  const wallOpts = { isStatic: true, restitution: 0.5, friction: 0 };
  const wallBodies = map.walls.map(w =>
    Bodies.rectangle(w.x, w.y, w.w, w.h, wallOpts)
  );
  World.add(world, wallBodies);

  const playerOpts = { restitution: 0.5, friction: 0, frictionAir: 0.1, label: 'player' };
  const player1 = Bodies.circle(map.playerStarts[0].x, map.playerStarts[0].y, C.PLAYER_RADIUS, playerOpts);
  const player2 = Bodies.circle(map.playerStarts[1].x, map.playerStarts[1].y, C.PLAYER_RADIUS, playerOpts);

}

function checkGoal(ball, map) {
  for (const goal of map.goals) {
    const inYRange = ball.position.y > goal.yMin && ball.position.y < goal.yMax;
    if (!inYRange) continue;
    if (goal.side === 'left'  && ball.position.x < 0)          return goal.scoringPlayer;
    if (goal.side === 'right' && ball.position.x > map.width)   return goal.scoringPlayer;
  }
  return null;
}

function startGameLoop(io, session, map) {
  const { engine, player1, player2, ball } = createPhysics(map);
  session.physics = { engine, player1, player2, ball };

  session.interval = setInterval(() => {
    if (session.scoring) return;

    applyInputs(player1, session.inputs[1] || {});
    applyInputs(player2, session.inputs[2] || {});
    stepWorld(engine, TICK_MS);

    const scorer = checkGoal(ball, map);
    if (scorer) {
      session.scoring = true;
      session.score[scorer]++;

      const winner = session.score[scorer] >= C.WIN_SCORE ? scorer : null;
      io.to(session.roomId).emit('goal', { scorer, score: { ...session.score }, winner });

      if (winner) {
        stopGameLoop(session);
        return;
      }

      setTimeout(() => {
        resetPositions(map, player1, player2, ball);
        session.scoring = false;
      }, 1000);
      return;
    }

    const state = getState(player1, player2, ball);
    io.to(session.roomId).emit('state', { ...state, score: session.score });
  }, TICK_MS);
}

function stopGameLoop(session) {
  clearInterval(session.interval);
  session.interval = null;
}

function createPhysics(map) {
  const engine = Engine.create({ gravity: { x: 0, y: 0 } });
  const world = engine.world;

  const wallOpts = { isStatic: true, restitution: 0.5, friction: 0 };
  const wallBodies = map.walls.map(w =>
    Bodies.rectangle(w.x, w.y, w.w, w.h, wallOpts)
  );
  World.add(world, wallBodies);

  const playerOpts = { restitution: 0.5, friction: 0, frictionAir: 0.1, label: 'player' };
  const player1 = Bodies.circle(map.playerStarts[0].x, map.playerStarts[0].y, C.PLAYER_RADIUS, playerOpts);
  const player2 = Bodies.circle(map.playerStarts[1].x, map.playerStarts[1].y, C.PLAYER_RADIUS, playerOpts);

  const ballOpts = { restitution: 0.8, friction: C.BALL_FRICTION, frictionAir: 0.005, label: 'ball' };
  const ball = Bodies.circle(map.ballStart.x, map.ballStart.y, C.BALL_RADIUS, ballOpts);

  World.add(world, [player1, player2, ball]);

  return { engine, player1, player2, ball };
}

function applyInputs(body, input, boosting) {
  const force = C.PLAYER_SPEED * (boosting ? C.BOOST_FORCE_MULTIPLIER : 1);
  if (input.up)    Body.applyForce(body, body.position, { x: 0,      y: -force });
  if (input.down)  Body.applyForce(body, body.position, { x: 0,      y:  force });
  if (input.left)  Body.applyForce(body, body.position, { x: -force, y: 0 });
  if (input.right) Body.applyForce(body, body.position, { x:  force, y: 0 });
}

function stepWorld(engine, delta) {
  Engine.update(engine, delta);
}

function checkGoal(ball, map) {
  for (const goal of map.goals) {
    const inYRange = ball.position.y > goal.yMin && ball.position.y < goal.yMax;
    if (!inYRange) continue;
    if (goal.side === 'left'  && ball.position.x < 0)          return goal.scoringPlayer;
    if (goal.side === 'right' && ball.position.x > map.width)   return goal.scoringPlayer;
  }
  return null;
}

let playerNumber = null;
let activeMap    = null;
let ctx          = null;
let socket       = null;

let latestScore  = { 1: 0, 2: 0 };
let latestBoosts = { 1: 100, 2: 100 };
let winner       = null;
let countdown    = null;

const TICK_MS = 1000 / CONSTANTS.TICK_RATE;

// Snapshot buffer — keeps the last several received states with timestamps.
// We render at a fixed delay behind the newest snapshot so we always have
// two frames to interpolate between, and the render time never resets.
const snapshots = [];
const RENDER_DELAY = TICK_MS * 2;  // render 2 ticks behind latest

socket = initNetwork(
  (num, map) => {
    playerNumber = num;
    activeMap = map;
    ctx = initRenderer(map);

    const color = num === 1 ? 'blue' : 'orange';
    document.getElementById('status').textContent =
      `You are Player ${num} (${color}). Waiting for opponent...`;
  },
  (state) => {
    const { score, boosts, ...gameState } = state;
    if (score)  latestScore  = score;
    if (boosts) latestBoosts = boosts;

    snapshots.push({ state: gameState, time: performance.now() });
    if (snapshots.length > 10) snapshots.shift();

    document.getElementById('status').textContent =
      `Player ${playerNumber} — WASD to move, Shift to boost`;
  },
  (scorer, score, w) => {
    latestScore = score;
    if (w) {
      winner = w;
      document.getElementById('status').textContent = `Player ${w} wins!`;
    } else {
      document.getElementById('status').textContent = `Player ${scorer} scored!`;
    }
  },
  (value) => {
    countdown = value;
    if (value === 'GO!') setTimeout(() => { countdown = null; }, 500);
  }
);

setInterval(() => {
  if (socket && playerNumber) socket.emit('input', getInput());
}, TICK_MS);

function lerp(a, b, t) {
  return a + (b - a) * t;
}

function getInterpolatedState() {
  if (snapshots.length < 2) return snapshots[0]?.state || null;

  // render time advances smoothly — never resets on new snapshot
  const renderTime = performance.now() - RENDER_DELAY;

  // find the two snapshots that bracket renderTime
  let prev = snapshots[0];
  let curr = snapshots[1];
  for (let i = 0; i < snapshots.length - 1; i++) {
    prev = snapshots[i];
    curr = snapshots[i + 1];
    if (curr.time >= renderTime) break;
  }

  const span = curr.time - prev.time;
  const t = span > 0 ? Math.max(0, Math.min(1, (renderTime - prev.time) / span)) : 1;

  return {
    p1:   { x: lerp(prev.state.p1.x,   curr.state.p1.x,   t), y: lerp(prev.state.p1.y,   curr.state.p1.y,   t) },
    p2:   { x: lerp(prev.state.p2.x,   curr.state.p2.x,   t), y: lerp(prev.state.p2.y,   curr.state.p2.y,   t) },
    ball: { x: lerp(prev.state.ball.x, curr.state.ball.x, t), y: lerp(prev.state.ball.y, curr.state.ball.y, t) },
  };
}

function gameLoop() {
  if (ctx && activeMap) {
    render(ctx, getInterpolatedState(), activeMap, latestScore, latestBoosts, winner, countdown);
  }
  requestAnimationFrame(gameLoop);
}

gameLoop();
